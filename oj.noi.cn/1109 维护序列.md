#### 答案  
1. 顺序表解法
```cpp
/*oj.noi.cn 1109 维护序列（顺序表解法）*/

 #include <iostream>
 using namespace std;
 
 //原数组最长1000，操作最多使长度增加1000（插入），所以新数组最长2000
 int m, n, a [2001];			
 
 //在第i位元素之前插入新元素，值为v
 void ins (int i, int v)	 
 {
 	for (int k=n; k>=i; k--)
 		a[k+1] = a[k];
 	a[i] = v;
 	n++;
 }

 //删除第i位元素
 void del (int i) 
 {
 	for (int k=i+1; k<=n; k++)
 		a[k-1] = a[k];
 	n--;
 }

 int main ()
 {
 	cin >> n;
 	for (int k=1; k<=n; k++)
 		cin >> a[k];
 	cin >> m;
 	
 	for (int k=1; k<=m; k++)
 	{
 		int type, i, v;
 		cin >> type >> i;
 		if (type==1)
 			cout << a[i] <<endl;
 		if (type==2)
 		{
 			cin >> v;
 			ins(i, v);
 		}
 		if (type==3)
 			del(i);
 	} 

 	return 0;
 }
 ```
